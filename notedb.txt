tabella users = [

    id INT(9) PRIMARY KEY AUTO_INCREMENT,
    username VARCHAR(30) NOT NULL,
    password VARCHAR(255) NOT NULL

];

tabella notes = [

    id INT(9) PRIMARY KEY AUTO_INCREMENT,
    utente_id INT(9) NOT NULL,
    title VARCHAR(20) NOT NULL,
    nota TEXT,
    FOREIGN KEY (utente_id) REFERENCES users(id)

];
